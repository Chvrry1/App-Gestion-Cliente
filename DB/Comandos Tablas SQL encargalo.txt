CREATE DEFINER=`root`@`localhost` PROCEDURE `spEncargoEnCamino` (IN `IdRepartidor` INT, IN `IdOrden` INT)  
BEGIN
    DECLARE errno INT;
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        GET CURRENT DIAGNOSTICS CONDITION 1 errno = MYSQL_ERRNO;
        SELECT errno AS MYSQL_ERROR;
        ROLLBACK;
    END;
    
    START TRANSACTION;
    
    SET @estadoRepartidor = 'OCUPADO'; 
    SET @estadoOrden = 4;

    # Actualizaci칩n del estado del repartidor
    UPDATE tblrepartidor SET repEstado = @estadoRepartidor
    WHERE idRepartidor = IdRepartidor;

    # Actualizaci칩n del estado de la orden
    UPDATE tblorden SET tblorden.odEstado = @estadoOrden
    WHERE tblorden.idOrden = IdOrden;
    
    COMMIT WORK;
END$$
